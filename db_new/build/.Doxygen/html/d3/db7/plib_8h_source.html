323/* PL_A is a wrapper that allows you to enter the struct
324 * data directly into a function without needing to define
325 * the type of struct, here is the before and after:
326 *
327 * int ret = pl_a((pl_arg){.name="test",.takes_value=1});
328 *
329 * int ret = PL_A(.name="test",.takes_value=1);

331 * so much easier right?

335/* to understand why this macro exists along with its implicit
336 * pl_arg_global_ptr. originally pl_a returned a pointer pl_arg
337 * to the actual item in the arguments array, this was very bad
338 * as when the argument list reached capacity it would overwrite
339 * the pointers memory location which creates a dangling pointer.
340 * this pointer can still be interacted with without error, the
341 * issue with this is that the pointer that we think is pointing
342 * to an argument in our argument list is actually not existant
343 * which means that from a debug point of view if you created an
344 * argument and the list was re-allocated then now for some
345 * reason regardless of how you trigger that argument it still
346 * comes up as false when using pl_arg_run this is of course
347 * because the pointer is refering to something other then what
348 * it is suppost too. to combat this i changed the code so an
349 * index is returned instead of a pointer, then after all of the
350 * args are set you can use either &PL_ARGS[index] or PL_GETI(i)
351 * to grab a pointer to an argument, now this was a good fix but
352 * now one line has turned into two lines, one for definition
353 * and a second for actually getting the pointer from the index.
354 * the way this is resolved is by using the following macro which
355 * is a copy of the PL_A macro but for the pl_arg_global_ptr func
356 * the pl_arg_global_ptr function will first check to see if the
357 * arg has been created if it has then it will search for it in
358 * the argument list and return a pointer to it if it finds it.
359 * if the arg has not been made then the arg will be offloaded
360 * into the pl_a function and then returned as a pointer. this
361 * means that if you initialize a var using the PL_P function
362 * every time you call the variable it will perform a search in
363 * the argument list for the argument which removes all risk of
364 * dangling pointers as the information is updated per-call and
365 * isent static and overwritable. */
366#define PL_P(...) pl_arg_global_ptr((pl_arg){__VA_ARGS__})
367pl_arg *pl_arg_global_ptr(pl_arg in);
